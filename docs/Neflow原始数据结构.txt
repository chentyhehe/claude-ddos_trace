create table uniflow_controller_clickhouse_trunk.analytics_netflow_dist
(
    app_rcv_time                 DateTime comment 'App收到Flow对象的时间',
    flow_ip_addr                 String comment '路由器的Flow IP地址',
    flow_sample_rate             UInt32 comment 'Flow采样比',
    src_ip_addr                  String comment '源IP地址',
    dst_ip_addr                  String comment '目的IP地址',
    next_hop_ip_addr             String comment '下一跳IP地址',
    input_if_index               UInt32 comment '入接口索引',
    output_if_index              UInt32 comment '出接口索引',
    octets                       UInt64 comment '乘以Flow采样比后的字节数',
    packets                      UInt64 comment '乘以Flow采样比后的包数',
    src_port                     UInt16 comment '源端口',
    dst_port                     UInt16 comment '目的端口',
    tcp_flags                    Int16 comment 'TCP标识',
    protocol                     UInt8 comment '协议号',
    tos                          UInt8 comment '服务类型',
    src_as                       UInt32 comment '源AS编号',
    dst_as                       UInt32 comment '目的AS编号',
    src_mask                     UInt8 comment '源IP地址掩码/前缀长度',
    dst_mask                     UInt8 comment '目的IP地址掩码/前缀长度',
    first_time                   UInt64 comment '开始时间',
    last_time                    UInt64 comment '结束时间',
    ip_ver                       UInt8 comment 'IP版本',
    service_dir                  String comment '业务方向',
    src_mo_name                  String comment '源监测对象名称',
    src_mo_code                  String comment '源监测对象编码',
    src_country                  String comment '源国家',
    src_province                 String comment '源省份',
    src_city                     String comment '源城市',
    src_isp                      String comment '源运营商',
    src_app                      String comment '源应用',
    input_net_boundary_id        String comment '入方向网络边界ID',
    input_if_match_net_boundary  UInt8 comment '入方向是否匹配网络边界',
    src_ip_match_attack          UInt8 comment '源IP是否匹配监测对象攻击检测的范围',
    src_ip_match_analysis        UInt8 comment '源IP是否匹配监测对象流量分析的范围',
    dst_mo_name                  String comment '目的监测对象名称',
    dst_mo_code                  String comment '目的监测对象编码',
    dst_country                  String comment '目的国家',
    dst_province                 String comment '目的省份',
    dst_city                     String comment '目的城市',
    dst_isp                      String comment '目的运营商',
    dst_app                      String comment '目的应用',
    output_net_boundary_id       String comment '出方向网络边界ID',
    output_if_match_net_boundary UInt8 comment '出方向是否匹配网络边界',
    dst_ip_match_attack          UInt8 comment '目的IP是否匹配监测对象攻击检测的范围',
    dst_ip_match_analysis        UInt8 comment '目的IP是否匹配监测对象流量分析的范围',
    flow_parse_id                String comment 'Flow解析单元ID',
    flow_version                 UInt8 comment 'Flow版本',
    flow_source_id               UInt32 comment 'Flow源ID',
    flow_sequence                UInt32 comment 'Flow序列号',
    flow_count                   UInt16 comment 'Flow数量',
    parser_rcv_time              UInt64 comment '解析单元收到Flow的时间',
    app_rcv_latency_ms           Int32 comment 'App接收Flow的毫秒延迟'
)
    engine = ReplicatedMergeTree('/clickhouse/uniflow_controller_clickhouse_trunk/tables/{shard}/analytics_netflow',
             '{replica}')
        PARTITION BY toYYYYMMDD(app_rcv_time)
        ORDER BY app_rcv_time
        TTL app_rcv_time + toIntervalMonth(12)
        SETTINGS index_granularity = 8192;
