CREATE TABLE `system_base_monitor_object` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '监测对象id',
  `code` varchar(36) DEFAULT NULL COMMENT '对象编码。目前必须等于id字段，否则存在业务功能问题。',
  `name` varchar(255) DEFAULT NULL COMMENT '对象名称',
  `description` text COMMENT '对象描述',
  `type` varchar(20) DEFAULT '' COMMENT '对象类型',
  `province` varchar(255) DEFAULT NULL COMMENT '省份',
  `city` varchar(255) DEFAULT NULL COMMENT '城市',
  `level` varchar(20) DEFAULT NULL COMMENT '等级 ',
  `bandwidth` int(11) DEFAULT NULL COMMENT '带宽',
  `bandwidth_unit` varchar(6) DEFAULT NULL COMMENT '带宽单位。取值：bps、Kbps、Mbps、Gbps、Tbps、Pbps。',
  `match_condition_type` varchar(255) DEFAULT NULL COMMENT '匹配条件类型，多个条件，地址段，接口，接口组，asn，应用id，aspath。ipv6，flow过滤条件，bgp团体，bgp扩展团体',
  `netboundary_id` varchar(2048) DEFAULT NULL COMMENT '绑定边界内容：接口组，路由器，接口的id',
  `threshold_enable` varchar(10) DEFAULT NULL COMMENT '阈值开关 ON / OFF',
  `threshold_id` bigint(20) DEFAULT NULL COMMENT '阈值ID。',
  `ip_prefix_enable` varchar(10) DEFAULT NULL COMMENT '阈值开关 ON / OFF',
  `ip_prefix_threshold_id` bigint(20) DEFAULT NULL COMMENT '阈值ID。',
  `ip_prefix_v4` int(11) DEFAULT NULL COMMENT 'IPv4扫段检测的掩码值 ',
  `ip_prefix_v6` int(11) DEFAULT NULL COMMENT 'IPv4扫段检测的掩码值 ',
  `ip_prefix_active_ip_num` int(11) DEFAULT NULL COMMENT 'ip扫段时ip段内IP地址数目 ',
  `create_time` timestamp NULL DEFAULT NULL COMMENT '创建时间',
  `update_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `abnormal_enable` varchar(50) DEFAULT NULL COMMENT '异常流量检测开关',
  `in_baseline_id` varchar(64) DEFAULT NULL COMMENT '入向异常流量检测基线模版',
  `out_baseline_id` varchar(64) DEFAULT NULL COMMENT '出向异常流量检测基线模版',
  `initiative_flow_report` varchar(4) DEFAULT NULL COMMENT '主动上报Flow开关。取值范围：ON/OFF',
  `important_custom` int(11) DEFAULT '0' COMMENT '1 是骨干大客户 0 否',
  `report_interval` int(11) DEFAULT '5' COMMENT '实时流量上报频率（单位：分钟），默认5分钟',
  `report_data_types` varchar(100) DEFAULT '' COMMENT '实时流量保障上报数据类型，多个用逗号分隔',
  PRIMARY KEY (`id`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=103695 DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC COMMENT='监测对象';
